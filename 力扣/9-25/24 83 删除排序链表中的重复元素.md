# 24 83 删除排序链表中的重复元素

```java
class Solution {
    public ListNode deleteDuplicates(ListNode head) {

        ListNode cur = head;

        while(cur!=null && cur.next!=null){

            if(cur.val == cur.next.val){
                cur.next = cur.next.next;
            }else{
                cur = cur.next;
            }
            
        }
        return head;
    }
}

class Solution {
    public ListNode deleteDuplicates(ListNode head) {

        ListNode cur = head;
		//只要当前节点和当前节点的下一个节点有值，就不断访问下去
        while(cur!=null && cur.next!=null){
			//当前和下一个相同，跳过下一个，让cur的next指向cur的下下一个
            if(cur.val == cur.next.val){
                cur.next = cur.next.next;
            //不相同，指针后移，cur后移一位，cur的next也后移一位，12->23
            }else{
                cur = cur.next;
            }
            
        }
        return head;
    }
}
```
